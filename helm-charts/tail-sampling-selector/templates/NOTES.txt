   kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "tail-sampling-selector.fullname" . }} {{ .Values.service.port }}:http &

   - Health:    http://localhost:{{ .Values.service.port }}/health
   - Ready:     http://localhost:{{ .Values.service.port }}/health/ready
   - Live:      http://localhost:{{ .Values.service.port }}/health/live
   - Metrics:   http://localhost:{{ .Values.service.metricsPort | default 9090 }}/metrics
